create database gerenciadordelivros;

create table gerenciadordelivros.usu_usuario(
	usu_id int not null auto_increment,
	usu_login varchar (30) not null,
	usu_senha varchar (30) not null,
	usu_status varchar (30) not null,
	usu_tipo varchar (30) not null,
	primary key (usu_id));
    
create table gerenciadordelivros.liv_livro(
	liv_id int not null auto_increment,
	liv_titulo varchar(255) not null,
	liv_autor varchar (255) not null,
	liv_editora varchar (255) not null,
	primary key (liv_id));

create table gerenciadordelivros.edi_editora(
	edi_id int not null auto_increment,
	edi_nome varchar(255) not null,
	edi_fundacao varchar (255) not null,
	edi_sede varchar (255) not null,
	edi_cnpj varchar (255) not null,
	primary key (edi_id));

create table gerenciadordelivros.aut_autor(
	aut_id int not null auto_increment,
	aut_nome varchar(255) not null,
	aut_idade varchar (255) not null,
	aut_editora varchar (255) not null,
	aut_genero varchar (255) not null,
	primary key (aut_id));

create table gerenciadordelivros.livro_autor (
id int NOT NULL AUTO_INCREMENT,
idLivro int,
idAutor int,
observacao VARCHAR(255),
primary key (id));

ALTER TABLE gerenciadordelivros.livro_autor ADD CONSTRAINT FK_AUT_LIV_ID FOREIGN KEY (IDAUTOR) REFERENCES gerenciadordelivros.aut_autor (aut_id);
ALTER TABLE gerenciadordelivros.livro_autor ADD CONSTRAINT FK_LIV_AUT_ID FOREIGN KEY (IDLIVRO) REFERENCES gerenciadordelivros.liv_livro (liv_id);
